<!DOCTYPE html>

<html>

<head>

    <title>Кот и мышь</title>

    <style>

        #cat {

            position: absolute;

            width: 50px;

            height: 50px;

            background-color: orange;

            border-radius: 50%;

        }


        #mouse {

            position: absolute;

            width: 20px;

            height: 20px;

            background-color: gray;

            border-radius: 50%;

        }

    </style>

</head>

<body>

    <div id="cat"></div>

    <div id="mouse"></div>


    <script>

        const cat = document.getElementById('cat');

        const mouse = document.getElementById('mouse');


        const gameAreaWidth = window.innerWidth;

        const gameAreaHeight = window.innerHeight;


        let catX = gameAreaWidth / 2;

        let catY = gameAreaHeight / 2;

        let mouseX = 0;

        let mouseY = 0;


        const catSpeed = 5;


        function updateMousePosition(event) {

            mouseX = event.clientX;

            mouseY = event.clientY;

        }


        function updateCatPosition() {

            if (Math.abs(catX - mouseX) > 2) {

                if (catX < mouseX) {

                    catX += catSpeed;

                } else {

                    catX -= catSpeed;

                }

            }


            if (Math.abs(catY - mouseY) > 2) {

                if (catY < mouseY) {

                    catY += catSpeed;

                } else {

                    catY -= catSpeed;

                }

            }

        }


        function gameLoop() {

            updateCatPosition();

            cat.style.left = catX + 'px';

            cat.style.top = catY + 'px';


            requestAnimationFrame(gameLoop);

        }


        document.addEventListener('mousemove', updateMousePosition);

        gameLoop();

    </script>

</body>

</html>
